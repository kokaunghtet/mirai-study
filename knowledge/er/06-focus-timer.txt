// Slide: Focus Timer
// Tables: pomodoro_settings, timer_sessions

// stub
Table users {
  id bigint [pk]
  username varchar
  display_name varchar
}

Table pomodoro_settings {
  id bigint [pk, increment]
  user_id bigint [not null, unique]
  focus_minutes integer [not null, default: 25]
  short_break_minutes integer [not null, default: 5]
  long_break_minutes integer [not null, default: 15]
  sessions_before_long_break integer [not null, default: 4]
  daily_goal_sessions integer [not null, default: 8]
  created_at timestamp
  updated_at timestamp
}

Table timer_sessions {
  id bigint [pk, increment]
  user_id bigint [not null]
  planned_duration integer [not null]
  actual_duration integer [null]
  completed boolean [not null, default: false]
  started_at timestamp [null]
  ended_at timestamp [null]
  created_at timestamp [default: `CURRENT_TIMESTAMP`]

  Indexes { user_id, started_at }
}

Ref: pomodoro_settings.user_id > users.id [delete: cascade]
Ref: timer_sessions.user_id > users.id [delete: cascade]
