// Slide: Moderation & Bans
// Tables: reports, user_bans, appeals, notifications

Enum report_target_type { post, user, comment }
Enum report_category { spam, harassment, misinformation, inappropriate, other }
Enum report_status { pending, reviewed, resolved, rejected }
Enum report_action_taken {
  removed_content
  temp_banned
  perm_banned
  temp_banned_removed
  perm_banned_removed
  none
}
Enum ban_type { temporary, permanent }
Enum appeal_status { pending, approved, rejected }
Enum notification_type {
  like_post
  comment_post
  follow_user
  system
  report_reviewed
  temp_ban
}

// stub
Table users {
  id bigint [pk]
  username varchar
  display_name varchar
}

Table reports {
  id bigint [pk, increment]
  reporter_id bigint [not null]
  target_type report_target_type [not null]
  target_id bigint [not null]
  category report_category [not null]
  reason text [null]
  mod_report boolean [not null, default: false]
  status report_status [not null, default: 'pending']
  reviewed_by bigint [null]
  action_taken report_action_taken [null]
  created_at timestamp
  updated_at timestamp

  Indexes { reporter_id, status, mod_report }
}

Table user_bans {
  id bigint [pk, increment]
  user_id bigint [not null]
  type ban_type [not null]
  reason text [null]
  report_id bigint [null]
  banned_by bigint [null]
  expires_at timestamp [null, note: 'null = permanent']
  lifted_at timestamp [null]
  lifted_by bigint [null, note: 'null = auto-expired']
  created_at timestamp
  updated_at timestamp

  Indexes { (user_id, lifted_at) }
}

Table appeals {
  id bigint [pk, increment]
  user_ban_id bigint [not null]
  user_id bigint [not null]
  message text [not null]
  status appeal_status [not null, default: 'pending']
  reviewed_by bigint [null]
  reviewed_at timestamp [null]
  created_at timestamp
  updated_at timestamp

  Indexes { status }
}

Table notifications {
  id bigint [pk, increment]
  user_id bigint [not null]
  sender_id bigint [null]
  type notification_type [not null]
  title varchar [not null]
  content text [not null]
  url varchar [null]
  read_at timestamp [null]
  created_at timestamp [default: `CURRENT_TIMESTAMP`]

  Indexes { user_id, read_at }
}

Ref: appeals.user_id > users.id [delete: cascade]
