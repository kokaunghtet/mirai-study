// Slide: Admin & Audit Trail
// Tables: activity_logs, post_revisions

// stub
Table users {
  id bigint [pk]
  username varchar
  display_name varchar
  role varchar
}

// stub
Table posts {
  id bigint [pk]
  user_id bigint
  title varchar [null]
  content text [null]
}

Table activity_logs {
  id bigint [pk, increment]
  user_id bigint [null]
  action varchar [not null, note: 'role_changed | report_filed | paper_added | question_added']
  subject_type varchar [null]
  subject_id bigint [null]
  properties json [null]
  created_at timestamp [null]

  Indexes { created_at, (action, created_at) }
}

Table post_revisions {
  id bigint [pk, increment]
  post_id bigint [not null]
  editor_id bigint [not null]
  title varchar [null]
  content text [null]
  created_at timestamp [default: `CURRENT_TIMESTAMP`]
}

Ref: activity_logs.user_id > users.id [delete: set null]
Ref: post_revisions.post_id > posts.id [delete: cascade]
Ref: post_revisions.editor_id > users.id [delete: cascade]
