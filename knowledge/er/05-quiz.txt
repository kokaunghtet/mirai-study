// Slide: Quiz System
// Tables: questions, question_revisions, quiz_attempts, user_answers

Enum answer_option { A, B, C, D }

// stubs
Table users {
  id bigint [pk]
  username varchar
}

Table exam_categories {
  id bigint [pk]
  name varchar
}

Table exam_levels {
  id bigint [pk]
  category_id bigint
  code varchar
  name varchar
}

Table questions {
  id bigint [pk, increment]
  category_id bigint [not null]
  level_id bigint [null]
  section varchar [null, note: 'JLPT: kanji/vocab/grammar | ITPEC FE: technology/strategy | null if level has no sections']
  text text [not null]
  option_a text [not null]
  option_b text [not null]
  option_c text [not null]
  option_d text [not null]
  answer answer_option [not null]
  explanation text [null]
  created_at timestamp
  updated_at timestamp

  Indexes { category_id, level_id, (category_id, level_id, section) }
}

Table question_revisions {
  id bigint [pk, increment]
  question_id bigint [not null]
  editor_id bigint [not null]
  action varchar [not null, note: 'created | edited']
  created_at timestamp [default: `CURRENT_TIMESTAMP`]
}

Table quiz_attempts {
  id bigint [pk, increment]
  user_id bigint [not null]
  category_id bigint [not null]
  level_id bigint [null]
  section varchar [null]
  total_questions integer [not null]
  score integer [null]
  started_at timestamp [null]
  completed_at timestamp [null]
  created_at timestamp [default: `CURRENT_TIMESTAMP`]

  Indexes { user_id, category_id }
}

Table user_answers {
  id bigint [pk, increment]
  attempt_id bigint [not null]
  question_id bigint [not null]
  selected_answer answer_option [not null]
  is_correct boolean [not null]
  created_at timestamp [default: `CURRENT_TIMESTAMP`]

  Indexes { attempt_id, question_id }
}

Ref: questions.category_id > exam_categories.id [delete: cascade]

Ref: question_revisions.question_id > questions.id [delete: cascade]
Ref: question_revisions.editor_id > users.id [delete: cascade]
Ref: quiz_attempts.user_id > users.id [delete: cascade]
Ref: quiz_attempts.category_id > exam_categories.id [delete: cascade]
Ref: quiz_attempts.level_id > exam_levels.id [delete: set null]
Ref: user_answers.attempt_id > quiz_attempts.id [delete: cascade]
Ref: user_answers.question_id > questions.id [delete: cascade]
